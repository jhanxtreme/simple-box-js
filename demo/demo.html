Simple Box - Demo Modal



HTML

//TRIGGER POINT
<element data-modal-trigger="my-modal1">Open Modal</element>

//SOURCE MODAL
<div class="simple-box modal" data-modal-source="my-modal1"> <div class="content"> ...CONTENT GOES HERE... </div> </div>

JAVASCRIPT

$('.simple-box').simpleBox({ targetType : "modal" });

//Advance options
$('.simple-box').simpleBox({ targetType : "modal", css: { width: '50%', height: '50%', borderRadius: '.2em', padding: '1em', overflow: 'auto' } });
